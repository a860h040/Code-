/** * Google Apps Script Web App — Application Launcher * * Spreadsheet format: * Column A: App name * Column B: App URL * Column C: App image URL * * Row 1 contains headings. * Data starts on row 2. */ /* ========================================================= CONFIGURATION — UPDATE THESE TWO VALUES ========================================================= */ /** * The spreadsheet ID is the long value in the Google Sheet URL: * * https://docs.google.com/spreadsheets/d/SPREADSHEET_ID/edit */ const SPREADSHEET_ID = 'PASTE_SPREADSHEET_ID_HERE'; /** * This must exactly match the name of the sheet tab. * Example: Apps */ const SHEET_NAME = 'Apps'; /* ========================================================= WEB APP ENTRY POINT ========================================================= */ /** * Serves the Index.html page when the web-app URL is opened. * * @return {HtmlOutput} */ function doGet() { return HtmlService .createHtmlOutputFromFile('Index') .setTitle('App Launcher') .addMetaTag( 'viewport', 'width=device-width, initial-scale=1, maximum-scale=5' ); } /* ========================================================= SPREADSHEET DATA ========================================================= */ /** * Reads app information from the configured Google Sheet. * * This function is called from Index.html through: * google.script.run.getApps() * * @return {Object} An object containing success status and app data. */ function getApps() { try { validateConfiguration_(); const spreadsheet = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { throw new Error( 'The sheet named "' + SHEET_NAME + '" was not found. ' + 'Check the SHEET_NAME value in Code.gs.' ); } const lastRow = sheet.getLastRow(); // Row 1 contains headings, so a last row below 2 means no app data exists. if (lastRow < 2) { return { success: true, apps: [], message: 'No apps have been added to the spreadsheet yet.' }; } /* * Read columns A through C, beginning at row 2. * * getDisplayValues() returns strings exactly as they appear in * the spreadsheet. */ const rows = sheet .getRange(2, 1, lastRow - 1, 3) .getDisplayValues(); const apps = rows .map(function (row, index) { const appName = cleanText_(row[0]); const appUrl = cleanUrl_(row[1]); const imageUrl = cleanImageUrl_(row[2]); return { name: appName, url: appUrl, imageUrl: imageUrl, rowNumber: index + 2 }; }) .filter(function (app) { /* * Ignore completely empty rows. * * A row is considered completely empty when the app name, * app URL, and image URL are all empty. */ return Boolean(app.name || app.url || app.imageUrl); }) .filter(function (app) { /* * A card needs an app name and a valid destination URL. * * Rows that only contain an image, only a name, or an invalid * URL are excluded from the launcher. */ return Boolean(app.name && app.url); }) .map(function (app) { /* * Do not expose the spreadsheet row number to the browser. * Return only the information needed by the page. */ return { name: app.name, url: app.url, imageUrl: app.imageUrl }; }); return { success: true, apps: apps, message: apps.length ? apps.length + ' app' + (apps.length === 1 ? '' : 's') + ' loaded.' : 'No valid apps were found.' }; } catch (error) { console.error('getApps error:', error); return { success: false, apps: [], message: getSafeErrorMessage_(error) }; } } /* ========================================================= VALIDATION AND SANITIZATION ========================================================= */ /** * Confirms that the required configuration values were updated. * * @throws {Error} */ function validateConfiguration_() { if ( !SPREADSHEET_ID || SPREADSHEET_ID === 'PASTE_SPREADSHEET_ID_HERE' ) { throw new Error( 'The spreadsheet ID is missing. Update SPREADSHEET_ID near ' + 'the top of Code.gs.' ); } if (!SHEET_NAME || !String(SHEET_NAME).trim()) { throw new Error( 'The sheet name is missing. Update SHEET_NAME near the top of Code.gs.' ); } } /** * Cleans plain text read from the spreadsheet. * * HTML is not created from this value in the frontend. The frontend * inserts it using textContent, which prevents spreadsheet text from * being interpreted as HTML. * * @param {*} value Value from the spreadsheet. * @return {string} */ function cleanText_(value) { if (value === null || value === undefined) { return ''; } return String(value) .replace(/[\u0000-\u0008\u000B\u000C\u000E-\u001F\u007F]/g, '') .trim() .substring(0, 200); } /** * Validates an app destination URL. * * Only HTTPS and HTTP links are permitted. Dangerous schemes such * as javascript:, data:, file:, and vbscript: are rejected. * * @param {*} value URL from Column B. * @return {string} */ function cleanUrl_(value) { const url = cleanText_(value); if (!url || url.length > 2048) { return ''; } try { const parsedUrl = new URL(url); const protocol = parsedUrl.getProtocol().toLowerCase(); if (protocol !== 'https' && protocol !== 'http') { return ''; } return parsedUrl.toString(); } catch (error) { return ''; } } /** * Validates an image URL from Column C. * * Only HTTPS and HTTP image addresses are permitted. * * @param {*} value Image URL from the spreadsheet. * @return {string} */ function cleanImageUrl_(value) { return cleanUrl_(value); } /** * Returns a controlled error message to the browser. * * @param {*} error Error object. * @return {string} */ function getSafeErrorMessage_(error) { if (error && error.message) { return String(error.message).substring(0, 500); } return 'An unexpected error occurred while loading the app list.'; } App Launcher
