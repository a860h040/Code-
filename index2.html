/** * GOOGLE APPS SCRIPT APP LAUNCHER * * Google Sheet format: * Column A = App name * Column B = App link * Column C = App image URL * * Row 1 = headings * Data starts on row 2 */ /* ========================================================= UPDATE THESE TWO VALUES ========================================================= */ const SPREADSHEET_ID = 'PASTE_YOUR_SPREADSHEET_ID_HERE'; const SHEET_NAME = 'Apps'; /* ========================================================= WEB APP ========================================================= */ /** * Opens Index.html when the web-app URL is visited. */ function doGet() { return HtmlService .createHtmlOutputFromFile('Index') .setTitle('App Launcher') .addMetaTag( 'viewport', 'width=device-width, initial-scale=1, maximum-scale=5' ); } /* ========================================================= GET APPS FROM GOOGLE SHEETS ========================================================= */ /** * Called by Index.html using google.script.run. */ function getApps() { try { validateSettings_(); const spreadsheet = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { throw new Error( 'The sheet tab named "' + SHEET_NAME + '" was not found. Check SHEET_NAME in Code.gs.' ); } const lastRow = sheet.getLastRow(); // The sheet only contains headings or is completely empty. if (lastRow < 2) { return { success: true, apps: [], message: 'No app rows were found below the headings.' }; } // Read columns A, B, and C starting from row 2. const rows = sheet .getRange(2, 1, lastRow - 1, 3) .getDisplayValues(); const apps = []; rows.forEach(function (row, index) { const appName = cleanText_(row[0]); const appUrl = cleanHttpUrl_(row[1]); const imageUrl = cleanHttpUrl_(row[2]); // Ignore completely empty rows. const rowIsEmpty = !cleanText_(row[0]) && !cleanText_(row[1]) && !cleanText_(row[2]); if (rowIsEmpty) { return; } /* * A usable app requires: * 1. An app name * 2. A valid HTTP or HTTPS destination URL * * The image URL is optional. */ if (!appName || !appUrl) { console.log( 'Skipped spreadsheet row ' + (index + 2) + ': missing app name or valid URL.' ); return; } apps.push({ name: appName, url: appUrl, imageUrl: imageUrl }); }); return { success: true, apps: apps, message: apps.length + ' app' + (apps.length === 1 ? '' : 's') + ' loaded.' }; } catch (error) { console.error(error); return { success: false, apps: [], message: error && error.message ? error.message : 'An unexpected error occurred.' }; } } /* ========================================================= CONFIGURATION VALIDATION ========================================================= */ function validateSettings_() { if ( !SPREADSHEET_ID || SPREADSHEET_ID === 'PASTE_YOUR_SPREADSHEET_ID_HERE' ) { throw new Error( 'The spreadsheet ID has not been entered in Code.gs.' ); } if (!SHEET_NAME || !String(SHEET_NAME).trim()) { throw new Error( 'The sheet name has not been entered in Code.gs.' ); } } /* ========================================================= DATA CLEANING ========================================================= */ /** * Cleans app names and other text. */ function cleanText_(value) { if (value === null || value === undefined) { return ''; } return String(value) .replace( /[\u0000-\u0008\u000B\u000C\u000E-\u001F\u007F]/g, '' ) .trim() .substring(0, 500); } /** * Allows only HTTP and HTTPS links. * * This version deliberately avoids URL.getProtocol(), * which caused the previous code to reject every URL. */ function cleanHttpUrl_(value) { const url = cleanText_(value); if (!url || url.length > 2048) { return ''; } // Reject spaces inside the URL. if (/\s/.test(url)) { return ''; } // Only allow HTTP and HTTPS. if (!/^https?:\/\/.+/i.test(url)) { return ''; } return url; } /* ========================================================= TEST FUNCTION ========================================================= */ /** * Run this function manually from the Apps Script editor. * * The execution log will show exactly what getApps() returns. */ function testGetApps() { const result = getApps(); console.log(JSON.stringify(result, null, 2)); }
